func main() {
var ch chan int
var ch2 chan int
ch = make(chan int, 10)
ch2 = make(chan int, 10)
go func() {
for i := 0; i < 10; i++ {
ch <- i
ch2 <- i * i
time.Sleep(time.Second)
}
}()

for {
    select {
    case v := <-ch:
        fmt.Println(v)
    case v := <-ch2:
        fmt.Println(v)
    default:
        fmt.Println("get data timeout")
        time.Sleep(time.Second)
    }
}

}
